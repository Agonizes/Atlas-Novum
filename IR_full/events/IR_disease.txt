#MEASLES
province_event = {
	id = disease.1
	title = "disease.1.t"
	desc = "disease.1.d"
	picture = PLAGUE_eventPicture
	
	trigger = { 
		is_city = yes 
		has_global_flag = IR_dev_init
		NOT = { is_year = 1444 }
		OR = { has_building = town has_building = city has_building = metropolis }
		IR_disease_free = yes
	}
	
	mean_time_to_happen = {
		months = 2000 # TEMPORARY AS NEED TO MATH OUT WITH PROPER DISEASES
		modifier = {
			factor = 0.9
			has_building = city
		}
		modifier = {
			factor = 0.7
			has_building = metropolis
		}
	}
	
	option = { #
		name = "disease.1.a" 
		add_devastation = 10
		add_permanent_province_modifier = {
			name = measles
   			duration = 730
		}
		IR_low_pop_loss = yes
		IR_check_dev_variables = yes
		every_neighbor_province = {
			limit = { IR_disease_free = yes }
			random = { 
    			chance = 5 
    			add_devastation = 10
				add_permanent_province_modifier = {
					name = measles
   					duration = 730
				}
				IR_low_pop_loss = yes
				IR_check_dev_variables = yes
			}
		}
	}
}

#MEASLES
province_event = {
	id = disease.11
	title = "disease.11.t"
	desc = "disease.11.d"
	picture = PLAGUE_eventPicture
	
	trigger = { 
		is_city = yes 
		has_global_flag = IR_dev_init 
		NOT = { is_year = 1444 }
		OR = {
			devastation = 25
			has_province_modifier = famine_1
			has_province_modifier = famine_2
			has_province_modifier = famine_3
			has_province_modifier = famine_4
			has_province_modifier = fading_famine
			has_province_modifier = lingering_famine
		}
		IR_disease_free = yes
	}
	
	mean_time_to_happen = {
		months = 4000 
		modifier = {
			factor = 0.5
			any_neighbor_province = {
				OR = { has_building = town has_building = city has_building = metropolis }
				has_province_modifier = measles
			}
		}
		modifier = {
			factor = 0.75
			devastation = 25
			NOT = { devastation = 50 }
		}
		modifier = {
			factor = 0.5
			devastation = 50
			NOT = { devastation = 75 }
		}
		modifier = {
			factor = 0.25
			devastation = 75
			NOT = { devastation = 90 }
		}
		modifier = {
			factor = 0.1
			devastation = 90
		}
		modifier = {
			factor = 0.75
			has_province_modifier = famine_1
		}
		modifier = {
			factor = 0.5
			has_province_modifier = famine_2
		}
		modifier = {
			factor = 0.25
			has_province_modifier = famine_3
		}
		modifier = {
			factor = 0.1
			has_province_modifier = famine_4
		}
		modifier = {
			factor = 0.9
			has_province_modifier = fading_famine
		}
		modifier = {
			factor = 0.5
			has_province_modifier = lingering_famine
		}
	}
	
	option = { #
		name = "disease.11.a" 
		add_devastation = 10
		add_permanent_province_modifier = {
			name = measles
   			duration = 730
		}
		IR_low_pop_loss = yes
		IR_check_dev_variables = yes
		if = { 
			limit = { NOT = { any_neighbor_province = { has_province_modifier = measles } } }
			every_neighbor_province = {
				limit = { IR_disease_free = yes }
				random = { 
    				chance = 50
    				add_devastation = 10
					add_permanent_province_modifier = {
						name = measles
   						duration = 730
					}
					IR_low_pop_loss = yes
					IR_check_dev_variables = yes
				}
			}
		}
	}
}
