				limit = { NOT = { country_or_non_sovereign_subject_holds = ROOT } }
				cede_province = ROOT 
			}

				west_prussia_area = {
					limit = { NOT = { country_or_non_sovereign_subject_holds = ROOT } }

					release_all_subjects = yes
				}
