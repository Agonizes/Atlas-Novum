#MISSION TRIGGERS
conquest_mission_trigger = {
	custom_trigger_tooltip = {
		tooltip = $TOOLTIP$
		NOT = {
			any_province = {
				$PROVINCES$ = yes
				NOT = { $OWNER_TYPE$ = ROOT }
			}
		}
		NOT = {
			any_province = {
				$PROVINCES$ = yes
				NOT = { is_city = yes }
			}
		}
	}
}

control_mission_trigger = {

				#}
			}
		}
	}
}

discover_mission_trigger = {
	custom_trigger_tooltip = {
		tooltip = $TOOLTIP$
		calc_true_if = {
			all_province = {
				$PROVINCES$ = yes
				has_discovered = ROOT
			}
			amount = $NUM$
		}
	}
}

num_colonized_mission_trigger = {
	custom_trigger_tooltip = {
		tooltip = $TOOLTIP$
		num_of_provinces_owned_or_owned_by_non_sovereign_subjects_with = {
			$PROVINCES$ = yes
			is_city = yes
			value = $NUM$
		}
	}
}

vassalize_mission_trigger = {
	custom_trigger_tooltip = {
		tooltip = $TOOLTIP$
		if = { 
			limit = { $TAG$ = { exists = yes } }
			$TAG$ = { 
				is_subject_of = ROOT 
				NOT = { is_subject_of_type = tributary_state }
			}
			NOT = {
				any_province = {
					is_core = $TAG$
					NOT = {	country_or_non_sovereign_subject_holds = ROOT }
				}
			}
		}
		else_if = {
			limit = { $TAG$ = { exists = no } }
			NOT = {
				any_province = {
					is_core = $TAG$
					NOT = {	owned_by = ROOT }
				}
			}
		}
	}
}

event_mission_trigger = {
	custom_trigger_tooltip = {
		tooltip = $TOOLTIP$
		OR = {
			has_country_flag = $EVENT$_happened
			is_year = $BYPASS_DATE$
		}
	}
}
