	variable_arithmetic_trigger = {
		tooltip = AN_MINORITY_CULTURE_CHECK
		12000 = {
			change_culture = $CULTURE$
			export_to_variable = {
				which = AN_minority_culture
				value = $CULTURE$
			}
			PREV = {
				is_variable_equal = { which = AN_minority_culture which = PREV } 
			}
		}
	}
}
