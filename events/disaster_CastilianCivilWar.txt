country_event = {
	id = castilian_succession.100
	title = "castilian_succession.100.t"
	desc = "castilian_succession.100.d"
	picture = CIVIL_WAR_eventPicture

	major = yes
	
	is_triggered_only = yes
	
	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = "castilian_succession.100.a" # hist
		trigger = {
			OR = {
				AND = {
					has_country_flag = cas_aragon_candidate
					is_year = 1480
					is_at_war = no
				}
			}
		}
		white_peace = POR
		white_peace = LON
		white_peace = FRA
		CAS = { inherit = LON }
		197 = { cede_province = FRA }
		add_opinion = {
			who = ARA
			modifier = cas_civil_war_winner
		}
		ARA = {
			add_opinion = {
				who = CAS
				modifier = cas_civil_war_winner
			}
		}
	}
	option = {
		name = "castilian_succession.100.b" # Portugal Triumphs
		trigger = {
			POR = { 
				overlord_of = LON 
			}
			CAS = { NOT = { controls = 219 } }
			ARA = { NOT = { controls = 220 } }
			POR = { controls = 227 }
			LON = { controls = 1745 }
		}
		white_peace = POR
		white_peace = LON
		white_peace = FRA
		197 = { cede_province = FRA }
		5096 = { cede_province = POR }
		366 = { cede_province = POR }
		#POR = { create_union = CAS }
		hidden_effect = {
			 every_province = {
				limit = { owned_by = LON }
				remove_core = ROOT
			}
		}
		if = {
			limit = {
				has_opinion_modifier = {
					modifier = cas_civil_war_interferance
					who = POR
				}
			}
			remove_opinion = {
				who = POR
				modifier = cas_civil_war_interferance
			}
		}
		add_opinion = {
			who = POR
			modifier = cas_civil_war_winner
		}
		POR = {
			add_opinion = {
				who = CAS
				modifier = cas_civil_war_winner
			}
		}
	}
	option = {
		name = "castilian_succession.100.c" # Castile Triumphs
		trigger = {
			OR = {
				CAS = { 
					overlord_of = LON 
					is_at_war = no 
				}
				AND = {
					POR = { NOT = { controls = 227 } }
					LON = { NOT = { controls = 1745 } }
					CAS = { controls = 219 }
					ARA = { controls = 220 }
				}
			}
		}
		white_peace = POR
		white_peace = LON
		white_peace = FRA
		CAS = { create_union = LON }
		POR = { release = CAS }
		if = {
			limit = { POR = { owns = 368 } }
			368 = { cede_province = CAS }
		}
		if = {
			limit = { POR = { owns = 334 } }
			334 = { cede_province = CAS }
		}
		if = {
			limit = { POR = { owns = 1751 } }
			1751 = { cede_province = CAS }
		}
		if = {
			limit = { POR = { owns = 5210 } }
			5210 = { cede_province = CAS }
		}
		if = {
			limit = { POR = { owns = 5218 } }

	}
