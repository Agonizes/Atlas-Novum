					region = oceanea_region

						region = california_region
						has_climate = arid
					}
					region = arabia_region
					province_id = 1118
					AND = {
						region = la_plata_region
						NOT = { has_climate = arid }
					}
					region = west_siberia_region
					region = east_siberia_region
					colonial_region = colonial_guyana
				}
				NOT = { 
					OR = {	
						area = northern_polynesia_area
					}
				}
			}
			add_permanent_province_modifier = {
				name = "soft_wasteland_2"
				duration = -1
			}
		}
		every_province = {
			limit = { 
				is_empty = yes

					region = guinea_region

					region = central_africa_region
					AND = {
						region = la_plata_region
						has_climate = arid
					}
					AND = {
						region = hudson_bay_region
						has_climate = arctic
					}
					AND = {
						region = canada_region
						has_climate = arctic
					}
					AND = {
						region = northwest_territories_region
						has_climate = arctic
					}
					AND = {
						region = alaska_region
						has_climate = arctic
					}
				}
				NOT = {
					OR = {
						province_id = 1174
						province_id = 1167
