#Sulawesi generally becomes Muslim
country_event = {
	id = IR_religion_convert.1
	title = IR_religion_convert.1.t
	desc = IR_religion_convert.1.d
	picture = COURT_eventPicture
	
	trigger = {
		capital_scope = { OR = { area = sulawesi_area area = south_sulawesi_area area = luwu_area area = banggai_area } NOT = { province_id = 644 } }
		is_year = 1500
		has_country_flag = IR_religion_convert.1_happened
	}

	mean_time_to_happen = {
		months = 480
		modifier = {
			tag = MKS
			factor = 0.25
		}
		modifier = {
			capital_scope = { area = east_sulawesi_area }
			factor = 0.1
		}
		modifier = {
			capital_scope = { area = donggala_area }
			factor = 0.5
		}
		modifier = {
			capital_scope = { area = south_sulawesi_area }
			factor = 0.75
		}
		modifier = {
			capital_scope = { area = sulawesi_area }
			factor = 1.25
		}
		modifier = {
			capital_scope = { area = luwu_area }
			factor = 1.5
		}
	}
	
	immediate = {
		set_country_flag = IR_religion_convert.1_happened
	}
	
	option = {
		name = IR_religion_convert.1.a #Accept the merchant's offer, convert
		ai_chance = {
			factor = 100
		}
		every_province = {
			limit = { OR = { province_id = 641 province_id = 11367 province_id = 2716 province_id = 11376 province_id = 6516 province_id = 2714 province_id = 6514 province_id = 11375 } owned_by = ROOT }
			change_religion = sunni
		}
		change_religion = sunni
		add_country_modifier = {
			name = "conversion_zeal"
			duration = 3650
		}
		every_subject_country = {
			limit = { NOT = { religion = sunni } }
			country_event = { id = IR_religion_convert.1 days = 90 random = 30 }
		}
	}
	option = {
		name = IR_religion_convert.1.b #Don't accept the merchant's offer
		ai_chance = {
			factor = 0
		}
		add_stability = -1
		add_prestige = 20
	}
}

#Islam in Mindanao
country_event = {
	id = IR_religion_convert.3
	title = IR_religion_convert.3.t
	desc = IR_religion_convert.3.d
	picture = COURT_eventPicture
	
	
	
	trigger = {
		capital_scope = { area = west_mindanao_area }
		is_year = 1515
		#MGD = { exists = no }
		has_country_flag = IR_religion_convert.3_happened
	}

	mean_time_to_happen = {
		months = 60
	}
	
	immediate = {
		set_country_flag = IR_religion_convert.3_happened
	}
	
	option = {
		name = IR_religion_convert.3.a #Accept the merchant's offer, convert
		ai_chance = {
			factor = 100
		}
		every_province = {
			limit = { mindanao_common_provinces = yes owned_by = ROOT }
			change_religion = sunni
			change_culture = bangsamoro
		}
		change_religion = sunni
		change_primary_culture = bangsamoro
		add_country_modifier = {
			name = "conversion_zeal"
			duration = 3650
		}
		every_subject_country = {
			limit = { NOT = { religion = sunni } }
			country_event = { id = IR_religion_convert.3 days = 90 random = 30 }
		}
		if = {
			limit = { MGD = { exists = no } 652 = { country_or_non_sovereign_subject_holds = ROOT num_of_cities = 2 } }
			change_tag = MGD
			hidden_effect = { set_capital = 652 }
		}
	}
	option = {
		name = IR_religion_convert.3.b #Don't accept the merchant's offer
		ai_chance = {
			factor = 0
		}
		add_stability = -1
		add_prestige = 20
	}
}
