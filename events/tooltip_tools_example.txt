namespace = tooltip_tools
country_event = {
	id = tooltip_tools.1
	title = ttt_test.1_t
	desc = ttt_test.1_d
	picture = event_test
	
	immediate = {
		hidden_effect = {
		
			#### EXAMPLE FOR TOOLTIP TOOL - BARS
			
			# do "event tooltip_tools.001" in the console to see

			### basic setup
			
			# id = bar ID, unique among its size, can be "1", "2", "3", "4" or "5"
			# size = bar size, can be "L" (event-width), "M" (regular bar width, "S" (modifier tooltip width) or "X" (tiny bar)
			# color = bar filling color, can be "1" (green), "2" (red), "3" (blue) or "4" (yellow)
			# bg = whether the bar should have a background or be transparent, can be "yes" or "no"
			
			# max = the value or variable at & above which the bar should be full
			# maxtype = whether "max" is a direct number or a variable name, can be "value" or "which"
			
			# min = the value or variable at & below which the bar should be empty
			# mintype = whether "min" is a direct number or a variable name, can be "value" or "which"
			
			# data = input value or variable that sets where the bar is at relative to the max and min values
			# datatype = whether "data" is a direct number or a variable name, can be "value" or "which"
			
			set_ttt_bar_data = { id=1 size=L color=1 bg=no data=25 datatype=value max=100 maxtype=value min=0 mintype=value }
			
			### shorter setup
			
			# combines size and ID - called "large", "medium", "small" or "tiny" and can be "1" to "5"
			
			# combines color and background - called "green", "red", "blue" or "yellow" and can be "yes" or "no"
			
			# combines max/min/data and maxtype/mintype/datatype, just add the type to the name
			
			set_ttt_bar_data = { large=2 red=yes datavalue=50 maxvalue=100 minvalue=0 }
			
			### example of variable usage
			
			set_variable = { which = input_var_1 value = 100 }
			set_variable = { which = input_var_2 which = input_var_1 }
			multiply_variable = { which = input_var_2 value = 0.33 }
			
			set_ttt_bar_data = { medium=1 blue=no datawhich=input_var_2 maxwhich=input_var_1 minvalue=0 }
			
			### some more to fill up the event
			
			set_ttt_bar_data = { medium=2 yellow=yes datavalue=100 maxvalue=100 minvalue=0 }
			set_ttt_bar_data = { small=1 green=no datavalue=66 maxvalue=100 minvalue=0 }
			set_ttt_bar_data = { small=2 red=yes datavalue=33 maxvalue=100 minvalue=0 }
			set_ttt_bar_data = { tiny=1 blue=no datavalue=1 maxvalue=100 minvalue=0 }
			set_ttt_bar_data = { tiny=2 yellow=yes datavalue=50 maxvalue=100 minvalue=0 }
		}
	}
	option = {
		name = ttt_test.1_o
	}
	after = {
	
		### clears relevant bar data, uses #size and #id or the combination, as above
	
		clear_ttt_bar_data = { medium=1 }
		clear_ttt_bar_data = { size=M id=2 }
		clear_ttt_bar_data = { id=1 size=S }
		clear_ttt_bar_data = { small=2 }
		clear_ttt_bar_data = { large=1 }
		clear_ttt_bar_data = { large=2 }
		clear_ttt_bar_data = { tiny=1 }
		clear_ttt_bar_data = { tiny=2 }
	}
}
