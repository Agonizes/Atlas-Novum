namespace = IR_reformation_centers

province_event = {
	id = IR_reformation_centers.1
	title = "IR_reformation_centers.1.t"
	desc = "IR_reformation_centers.1.d"
	picture = RELIGION_eventPicture
	hidden = yes
	
	trigger = {
		is_religion_enabled = reformed
		religion_years = { reformed = 5 }
		NOT = { religion_years = { protestant = 135 } }
		NOT = { current_age = age_of_absolutism }
		NOT = { current_age = age_of_revolutions }		
		NOT = {
			calc_true_if = {
				all_province = {
					religion = reformed
					is_reformation_center = yes
				}
				amount = 5
			}
		}
		owner = {
			capital_scope = {
				superregion = europe_superregion
			}
		}
		religion = reformed
		NOT = { 
			calc_true_if = {
				all_province = {
					religion = protestant
					is_reformation_center = yes
					culture = ROOT
				}
				amount = 1
			}
		}
	}
	
	mean_time_to_happen = {
		months = 60

		modifier = {
			factor = 0.1
			owner = {
				religion = reformed
			}
		}
	}
	
	option = {
		name = "IR_reformation_centers.1.a"
		add_reform_center = reformed
		add_permanent_province_modifier = {
			name = "religious_zeal_at_conv"
			duration = 9000
		}
	}
}

country_event = {
	id = IR_reformation_centers.3
	title = "IR_reformation_centers.3.t"
	desc = "IR_reformation_centers.3.d"
	hidden = yes
	
	trigger = {
		religion = anglican
		NOT = { current_age = age_of_absolutism }
		NOT = { current_age = age_of_revolutions }
		NOT = {
			calc_true_if = {
				all_province = {
					religion = anglican
					is_reformation_center = yes
				}
				amount = 1
			}
		}
		capital_scope = {
			region = british_isles_region
		}
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = {
		random_owned_province = {
			limit = {
				region = british_isles_region
				is_capital = yes
				culture = ROOT
			}
			add_reform_center = anglican
			change_religion = anglican
			add_permanent_province_modifier = {
				name = "religious_zeal_at_conv"
				duration = 9000
			}
		}
	}
}

	}

			limit = {

				is_reformation_center = yes	
			}
			remove_reform_center = anglican
		}
	}
}
