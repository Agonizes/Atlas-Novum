namespace = IR_flavor_greek
	
# Split of Greek once Byz is gone
country_event = {
	id = IR_flavor_greek.1
	title = "IR_flavor_greek.1.t"
	desc = "IR_flavor_greek.1.d"
	picture = COURT_eventPicture

	fire_only_once = yes
	major = yes

	trigger = {
		BYZ = { exists = no }
		any_province = {
			culture = greek
		}
		151 = { culture = turkish }
		NOT = { has_global_flag = IR_flavor_greek.1_happened }
	}

	mean_time_to_happen = {
		months = 120
	}
	
	
	immediate = {
		set_global_flag = IR_flavor_greek.1_happened
	}
	
	
	option = {		#Convert
		every_province = {
			limit = { culture = greek }
			if = {
				limit = { OR = { area = thrace_area area = northern_thrace_area area = upper_thrace_area } }
				change_culture = thracian
			}
			if = {
				limit = { OR = { area = macedonia_area area = eastern_macedonia_area } }
				change_culture = makedon
			}
			if = {
				limit = { OR = { area = thessaly_area province_id = 10131 province_id = 7113 } }
				change_culture = thessalian
			}
			if = {
				limit = { OR = { area = western_aegean_area province_id = 7112 province_id = 7108 } NOT = { province_id = 7113 } }
				change_culture = hellenic
			}
			if = {
				limit = { OR = { area = aegean_archipelago_area area = south_aegean_area province_id = 10134 } NOT = { province_id = 10131 } }
				change_culture = mikrasiatic
			}
		}
	}
}
