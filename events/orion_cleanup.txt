country_event = { 
	id = cleanup.3
	title = "cleanup.3.t"
	desc = "cleanup.3.d"
	picture = FREE_CITY_eventPicture
	
	hidden = yes
	mean_time_to_happen = { 
		days = 1
		modifier = {
			factor = 0.001
			is_subject = yes
			is_colony = no
			is_lesser_in_union = no
			NOT = { is_subject_of_type = fief }
			NOT = { is_subject_of_type = canton }
			NOT = { is_subject_of_type = dominion }
			NOT = { is_subject_of_type = league }
			NOT = { num_of_cities = 9 }
		}	
	}

	trigger = {
		ai = yes
		is_subject = yes
		is_colony = no
		is_lesser_in_union = no
		NOT = { is_subject_of_type = fief }
		NOT = { is_subject_of_type = canton }
		NOT = { is_subject_of_type = dominion }
		NOT = { is_subject_of_type = league }
		OR = {
			technology_group = western
			technology_group = eastern
			technology_group = ottoman
			technology_group = muslim
		}
		is_at_war = no
		NOT = { num_of_cities = 9 }
		overlord = { 
			religion_group = ROOT
			culture_group = ROOT
			OR = {
				technology_group = western
				technology_group = eastern
				technology_group = ottoman
				technology_group = muslim
			}
		}
    }

	option = {
		name = "cleanup.3.a"
		ai_chance = { factor = 100 }		
		overlord = {
			country_event = { id = cleanup.4 days = 0 }
		}
	}
}
