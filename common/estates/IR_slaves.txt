estate_slaves = {
	icon = 5
	
	color = { 46 63 68 }
	
	# Is estate kept in control of province on conquest?
	keep_on_conquest = no	

	# Giving territory affects power this much compared to base
	territory_power_modifier = 0.5	

	# If true, country will get estate
	trigger = {	
		OR = {
			is_colonial_nation = yes
			AND = {
				capital_scope = { continent = europe }
				any_owned_province = {
					trade_goods = slaves 
				}
			}
		}
		OR = {
			technology_group = western
			technology_group = eastern
			technology_group = ottoman
		}
	}
	
	# If true, province can be granted to estate
	province_trigger = {
		#owner = { government = colonial_government }
		OR = { 
			trade_goods = grain
			trade_goods = copper
			trade_goods = gold
			trade_goods = iron
			trade_goods = tea
			trade_goods = spices
			trade_goods = coffee
			trade_goods = cotton
			trade_goods = sugar
			trade_goods = tobacco
			trade_goods = cocoa
			trade_goods = dyes
			trade_goods = gems
			trade_goods = coal
		}
		NOT = { continent = europe }
	}	
	
	# Min autonomy in estate provinces
	min_autonomy = 25
	
	# Estate cancels out the following multiplicative local autonomy effects:
	autonomy_does_not_affect = local_production_efficiency
	autonomy_does_not_affect = province_trade_power_modifier

	# These scale with loyalty & power
	country_modifier_happy = {
		production_efficiency = 0.25
		development_cost = -0.25
	}
	country_modifier_neutral = {
		production_efficiency = 0.1
	}	
	country_modifier_angry = {
		production_efficiency = -0.25
		development_cost = 0.25
	}
	
	# These do not scale, but only applied in provinces owned by the estate
	province_modifier_happy = {
		trade_goods_size = 1
		local_build_cost = -0.2
	}
	province_modifier_neutral = {
		trade_goods_size = 0.5
		local_build_cost = -0.1
	}		
	province_modifier_angry = {
		trade_goods_size = -1
		local_build_cost = 0.2
		local_unrest = 25	
	}	
	
	# Influence modifiers
	base_influence = 0
	influence_modifier = {
		desc = EST_HIGH_DEVELOPMENT_PROVINCE
		trigger = {
			any_owned_province = {
				NOT = { religion_group = muslim }
			}
		}
		influence = 10		
	}
	influence_modifier = {
		desc = EST_VAL_RULED_BY_RELIGIOUS_MINORITY
		trigger = {
			NOT = { dominant_religion = ROOT }
			any_owned_province = {
				NOT = { religion = ROOT }
			}
		}
		influence = 20
	}
	influence_modifier = {	
		desc = EST_VAL_COT
		trigger = {
			NOT = { government = imperial_city }
			any_owned_province = {
				OR = {
					has_province_modifier = inland_center_of_trade_modifier #LOOK AT THIS!
					has_province_modifier = center_of_trade_modifier
				}
			}
		}
		influence = 5
	}
	influence_modifier = {
		desc = EST_VAL_OTHER_ESTATE_IN_POWER
		trigger = {
			OR = {
				has_disaster = estate_nobility_disaster
				has_disaster = estate_church_disaster
				has_disaster = estate_burghers_disaster
			}
		}
		influence = -40
	}
	
	#Loyalty Modifiers
	loyalty_modifier = {
		desc = EST_VAL_OTHER_ESTATE_IN_POWER_LOY
		trigger = {
			OR = {
				has_disaster = estate_nobility_disaster
				has_disaster = estate_church_disaster
				has_disaster = estate_burghers_disaster
			}
		}
		loyalty = -20
	}
	
	loyalty_modifier = {
		desc = EST_VAL_FULL_HUMANIST
		trigger = {
			full_idea_group = humanist_ideas
		}
		loyalty = -50
	}
	
	# Does the AI want to give any extra territory beyond the minimum?
	ai_territory_modifier = {
		trigger = {
			trigger = {
				government = monarchy
				OR = {
					technology_group = western
					technology_group = eastern
					technology_group = ottoman
				}
				num_of_cities = 10
			}	
			value = 10
		}	
		value = 100 # Code has a sanity check against pushing influence too high
	}	
	
	
	# For province assignment
	ai_grant_province = {
		factor = 1
		modifier = {
			factor = 10
			has_owner_religion = no
			tolerance_to_this = 1
		}
		modifier = {
			factor = 4
			local_autonomy = 25
		}
		modifier = {
			factor = 0.25
			NOT = { local_autonomy = 25 }
		}
		modifier = {
			factor = 10
			base_production = 5 
			OR = { 
				trade_goods = grain
				trade_goods = copper
				trade_goods = gold
				trade_goods = iron
				trade_goods = tea
				trade_goods = spices
				trade_goods = coffee
				trade_goods = cotton
				trade_goods = sugar
				trade_goods = tobacco
				trade_goods = cocoa
				trade_goods = dyes
				trade_goods = gems
				trade_goods = coal
			}
		}
	}	
	
	ai_grant_province = {
		factor = 1
		modifier = {
			factor = 0.5
			NOT = { province_trade_power = 5 }
		}			
		modifier = {
			factor = 10
			OR = { 
				trade_goods = grain
				trade_goods = copper
				trade_goods = gold
				trade_goods = iron
				trade_goods = tea
				trade_goods = spices
				trade_goods = coffee
				trade_goods = cotton
				trade_goods = sugar
				trade_goods = tobacco
				trade_goods = cocoa
				trade_goods = dyes
				trade_goods = gems
				trade_goods = coal
			}
		}
		modifier = {
			factor = 4
			local_autonomy = 25
		}
		modifier = {
			factor = 0.25
			NOT = { local_autonomy = 25 }
		}
	}	
	
	# Special interactions
	interaction = {
		name = "HARSHER_CONDITIONS"
		icon = 6
		
		years_between_use = 10
		
		potential = {
			NOT = { has_disaster = estate_church_disaster }
			NOT = { has_disaster = estate_nobility_disaster }
		}
		
		trigger = {
			estate_loyalty = {
				estate = estate_slaves
				loyalty = 30
			}				
		}
		
		effect = {
			if = {
				limit = {
					NOT = {
						estate_influence = {
							estate = estate_slaves
							influence = 50
						}	
					}
				}
				add_years_of_income = 0.25
			}
			if = {
				limit = {
					estate_influence = {
						estate = estate_slaves
						influence = 50
					}
					NOT = {
						estate_influence = {
							estate = estate_slaves
							influence = 75
						}	
					}
				}
				add_years_of_income = 0.5
			}
			if = {
				limit = {
					estate_influence = {
						estate = estate_slaves
						influence = 75
					}
					NOT = {
						estate_influence = {
							estate = estate_slaves
							influence = 100
						}	
					}
				}
				add_years_of_income = 0.75
			}
			if = {
				limit = {
					estate_influence = {
						estate = estate_slaves
						influence = 100
					}
				}
				add_years_of_income = 1
			}				
			add_estate_loyalty = {
				estate = estate_slaves
				loyalty = -15
			}
		}
		
		ai_will_do = {
			factor = 1
			modifier = {
				factor = 0
				NOT = {
					estate_loyalty = {
						estate = estate_slaves
						loyalty = 50
					}	
				}
			}
			modifier = {
				factor = 0
				treasury = 1
			}
		}
	}
	interaction = {
		name = "DEMAND_SUPPORT_SLAVE_OWNERS"
		icon = 3
		
		years_between_use = 20
		
		potential = {
			NOT = { has_disaster = estate_church_disaster }
			NOT = { has_disaster = estate_nobility_disaster }
		}
		
		trigger = {		
			estate_loyalty = {
				estate = estate_slaves
				loyalty = 30
			}	
			estate_influence = {
				estate = estate_slaves
				influence = 25
			}			
		}
		
		effect = {
			if = {
				limit = {
					NOT = {
						estate_influence = {
							estate = estate_slaves
							influence = 50
						}	
					}
				}
				add_adm_power = 50
			}
			if = {
				limit = {
					estate_influence = {
						estate = estate_slaves
						influence = 50
					}
					NOT = {
						estate_influence = {
							estate = estate_slaves
							influence = 75
						}	
					}
				}
				add_adm_power = 100
			}
			if = {
				limit = {
					estate_influence = {
						estate = estate_slaves
						influence = 75
					}
					NOT = {
						estate_influence = {
							estate = estate_slaves
							influence = 100
						}	
					}
				}
				add_adm_power = 150
			}
			if = {
				limit = {
					estate_influence = {
						estate = estate_slaves
						influence = 100
					}
				}
				add_adm_power = 200
			}				
			add_estate_loyalty = {
				estate = estate_slaves
				loyalty = -20
			}
		}
		
		ai_will_do = {
			factor = 0
		}
	}
	interaction = {
		name = "GRANT_AGRICULTURE_PLANTATIONS"
		icon = 17
		
		years_between_use = 20
		
		trigger = {
			always = yes
		}
		
		effect = {
			add_prestige = -5
			add_estate_influence_modifier = {
				estate = estate_slaves
				desc = EST_VAL_GRANTED_PLANTATIONS
				influence = 10
				duration = 7300
			}
			add_estate_loyalty = {
				estate = estate_slaves
				loyalty = 15
			}
		}
		
		ai_will_do = {
			factor = 1
			modifier = {
				factor = 0
				estate_influence = {
					estate = estate_slaves
					influence = 60
				}				
			}
			modifier = {
				factor = 0
				estate_loyalty = {
					estate = estate_slaves
					loyalty = 30
				}				
			}			
		}
	}
