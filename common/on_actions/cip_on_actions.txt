on_startup = {
	events = {
		muslim_school_events.20 #Pick School
	}
	if = {
		limit = {
			has_dlc = "Cradle of Civilization"
			has_country_modifier = tur_janissary
		}
		remove_country_modifier = tur_janissary
	}
	initialize_schools_effect = yes #This is only used here but is used for readability of on_action file.
	
	#New for better addon support
	if = {
		limit = { NOT = { has_global_flag = beyond_typus_mod_enabled } }
		set_global_flag = beyond_typus_mod_enabled
	}
	#Nobiscum Deus
	if = {
		limit = {
			NOT = { has_global_flag = nd_enabled }
		}
		set_global_flag = nd_enabled
		random_country = {			
			country_event = { id = nd_initial_setup.1 }
		}
	}
}
