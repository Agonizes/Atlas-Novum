IR_noport_forcelimit_1 = {
	potential = {
		NOT = { num_of_ports = 5 }
		NOT = {
			calc_true_if = {
				all_province = {
					country_or_non_sovereign_subject_holds = ROOT
					has_port = yes
				}
				amount = 10
			}
		}
	}
	trigger = {
		custom_trigger_tooltip = {
			tooltip = IR_noport_forcelimit_1.tooltip
			calc_true_if = {
				all_province = {
					country_or_non_sovereign_subject_holds = ROOT
					has_port = yes
				}
				amount = 5
			}
		}
	}
	naval_forcelimit = 5
	global_sailors = 625
}
