cb_fief_vassal = {
	valid_for_subject = no

	prerequisites = {
		current_age = age_of_discovery
		is_fief_domain_tech = yes
		is_subject = no
		has_institution = feudalism
		if = {
			limit = { is_part_of_hre = yes }
			FROM = {
				NOT = { num_of_cities = ROOT }
				NOT = { num_of_cities = 9 }
				NOT = { is_subject_of = ROOT }
				is_subject = no
				is_part_of_hre = yes
				is_elector = no
				NOT = { has_reform = imperial_city }
				is_emperor = no
				religion_group = ROOT
				is_neighbor_of = ROOT
				is_fief_domain_tech = yes
			}
		}
		else_if = {
			limit = { religion_group = muslim }
			FROM = {
				NOT = { num_of_cities = ROOT }
				NOT = { num_of_cities = 9 }
				NOT = { is_subject_of = ROOT }
				is_subject = no
				OR = {
					religion_group = ROOT
					culture_group = ROOT
				}
				is_neighbor_of = ROOT
				is_fief_domain_tech = yes
			}
		}
		else = {
			FROM = {
				NOT = { num_of_cities = ROOT }
				NOT = { num_of_cities = 9 }
				NOT = { is_subject_of = ROOT }
				is_subject = no
				culture_group = ROOT
				religion_group = ROOT
				is_neighbor_of = ROOT
				is_part_of_hre = no
				is_fief_domain_tech = yes
			}
		}
	}

	attacker_disabled_po = {
		po_demand_provinces
	#	po_revoke_cores
	#	po_release_vassals
	#	po_release_annexed
	#	po_return_cores

		po_become_tributary_state
		po_form_personal_union

	##	po_war_reparations
	    po_change_government
	    po_revoke_elector
	    po_trade_power
	    po_steer_trade
	    po_humiliate
	}
	
	war_goal = take_capital_fief
}
