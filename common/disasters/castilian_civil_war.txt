# War of Castilian Succession
# Railroaded succession crisis after Iberian wedding, which should trigger every single time
#
# By Amyzonis
#

castilian_succession = {
	potential = {
		tag = CAS
		NOT = { has_country_flag = had_cas_civil_war }
		is_free_or_tributary_trigger = yes
		exists = ARA
		exists = POR
		POR = { is_free_or_tributary_trigger = yes }
		government = monarchy
		current_age = age_of_discovery
		is_lesser_in_union = no
	}


	can_start = {
		is_at_war = no
		has_any_disaster = no
		senior_union_with = ARA
	}
	
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			is_lesser_in_union = yes
			NOT = { government = monarchy }
			is_year = 1480
		}
	}
	
	
	
	
	progress = {
		modifier = {
			factor = 3
			senior_union_with = ARA 
		}
	}




	
	can_end = {
		OR = {
			AND = {
				is_year = 1480
				is_at_war = no	
				LON = { junior_union_with = POR }	
			}
			AND = {
				is_at_war = no	
				LON = { junior_union_with = CAS }
			}
		}
	}
	



	modifier = {
		global_unrest = 5
	}	
	
	
	on_start = castilian_succession.1
	on_end = castilian_succession.100
	
	
}
