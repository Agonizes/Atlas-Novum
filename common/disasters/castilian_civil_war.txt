#

castilian_succession = {
	potential = {
		tag = CAS
		NOT = { has_country_flag = had_cas_civil_war }
		is_free_or_tributary_trigger = yes
		exists = ARA
		exists = POR
		POR = { is_free_or_tributary_trigger = yes }
		government = monarchy
		current_age = age_of_discovery
		is_lesser_in_union = no
	}


	can_start = {
		is_at_war = no
		has_any_disaster = no
		senior_union_with = ARA
	}
	
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			is_lesser_in_union = yes
			NOT = { government = monarchy }
			is_at_war = yes
			is_year = 1480
		}
	}
	
	
	
	
	progress = {
		modifier = {
			factor = 2
			if = { 
				limit = { ARA = { ai = yes exists = yes }
				senior_union_with = ARA
			}
			if = { 
				limit = { CAS = { ai = yes exists = yes }
				senior_union_with = CAS
			}
		}
	}




	
	can_end = {
		OR = {
			AND = {
				is_year = 1480
				is_at_war = no	
			}
		}
	}
	



	modifier = {
		global_unrest = 5
	}	
	
	
	on_start = castilian_succession.1
	on_end = castilian_succession.100
	
	on_monthly = {
		events = {
		}
		random_events = { 
			180 = 0
			1 = castilian_succession.2 #Civil War Spreads to Province
			#Events 3 - 8 related to foreign intervention
		}
	}
}
