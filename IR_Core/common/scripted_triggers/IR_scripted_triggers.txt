#MISSION TRIGGERS
country_or_non_sovereign_subject_controls = {
	custom_trigger_tooltip = {
		tooltip = country_or_non_sovereign_subject_controls.tooltip
		OR = {
			controlled_by = $TAG$
			controller = { is_subject_of = $TAG$ NOT = { is_subject_of_type = tributary_state } }
		}
	}
}

conquest_mission_trigger = {
	custom_trigger_tooltip = {
		tooltip = $TOOLTIP$
		NOT = {
			any_province = {
				$PROVINCES$ = yes
				NOT = { $OWNER_TYPE$ = ROOT }
			}
		}
		NOT = {
			any_province = {
				$PROVINCES$ = yes
				NOT = { is_city = yes }
			}
		}
	}
}

discover_mission_trigger = {
	custom_trigger_tooltip = {
		tooltip = $TOOLTIP$
		calc_true_if = {
			all_province = {
				$PROVINCES$ = yes
				has_discovered = ROOT
			}
			amount = $NUM$
		}
	}
}

num_colonized_mission_trigger = {
	custom_trigger_tooltip = {
		tooltip = $TOOLTIP$
		calc_true_if = {
			all_province = {
				$PROVINCES$ = yes
				is_city = yes
				country_or_non_sovereign_subject_holds = ROOT
			}
			amount = $NUM$
		}
	}
}


vassalize_mission_trigger = {
	custom_trigger_tooltip = {
		tooltip = $TOOLTIP$
		if = { 
			limit = { $TAG$ = { exists = yes } }
			$TAG$ = { 
				is_subject_of = ROOT 
				NOT = { is_subject_of_type = tributary_state }
			}
			NOT = {
				any_province = {
					is_core = $TAG$
					NOT = {	country_or_non_sovereign_subject_holds = ROOT }
				}
			}
		}
		else_if = {
			limit = { $TAG$ = { exists = no } }
			NOT = {
				any_province = {
					is_core = $TAG$
					NOT = {	owned_by = ROOT }
				}
			}
		}
	}
}

event_mission_trigger = {

			is_year = $BYPASS_DATE$
		}
	}
}
