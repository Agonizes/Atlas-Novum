# Revenge for Varna - TRIGGERS BY OTTOMAN MISSION 
# HUNGARY INITIATES

country_event = {
	id = IR_flavor_HUN.1
	title = "IR_flavor_HUN.1.t"
	desc = "IR_flavor_HUN.1.d"
	picture = COURT_eventPicture

	fire_only_once = yes
	major = yes

	trigger = {
		tag = HUN
		#is_year = 1447
		is_at_war = no
		NOT = { has_country_flag = IR_flavor_HUN.1_happened }
	}
	
	mean_time_to_happen = {
		months = 48
		modifier = {
			factor = 0.25
			is_year = 1446
		}
		modifier = {
			factor = 0.25
			TUR = { war_with = ALB }
			TUR = { NOT = { war_with = KAR } }
		}
	}
	
	
	immediate = {
		set_country_flag = IR_flavor_HUN.1_happened
	}
	
	
	option = {		# Assemble an invasion force - prompty Albania and Serbia
		name = "IR_flavor_HUN.1.a"
		ai_chance = { factor = 95 }
		set_country_flag = IR_flavor_HUN.1_a
		SER = { country_event = { id = IR_flavor_HUN.2 days = 5 } }
		ALB = { country_event = { id = IR_flavor_HUN.2 days = 10 } }
		TUR = { country_event = { id = IR_flavor_TUR.1 days = 30 } } # CHECKS SER AND ALB REACTION
	}
	option = {		# Abandon the lower balkans - path leads to rare chance to seek understanding with ottomans?
		name = "IR_flavor_HUN.1.b"
		ai_chance = { factor = 5 }
		set_country_flag = IR_flavor_HUN.1_b
	#	TUR = { country_event = { id = IR_flavor_TUR.1 days = 30 } } # NEEDS NEW EVENT
	}
}

# CHECK ALBANIA AND SERBIA FOR COALITION VS OTTOMANS
country_event = {
	id = IR_flavor_HUN.2
	title = "IR_flavor_HUN.2.t"
	desc = "IR_flavor_HUN.2.d"
	picture = COURT_eventPicture

	fire_only_once = yes
	is_triggered_only = yes

	trigger = {
		OR = {
			tag = ALB
			tag = SER
		}
		NOT = { has_country_flag = IR_flavor_HUN.2_happened }
	}
	
	
	
	immediate = {
		set_country_flag = IR_flavor_HUN.2_happened
	}
	
	
	option = {		# The Turkish Menace must be destroyed
		name = "IR_flavor_HUN.2.a"
		ai_chance = { 
			factor = 50
			modifier = {
				factor = 2
				tag = ALB
				war_with = TUR
			}
			modifier = {
				factor = 2
				tag = SER
				is_subject = no
			}
			modifier = {
				factor = 2
				tag = SER
				is_subject_of = TUR
				NOT = { has_opinion = {
					who = TUR
					value = -50
				} }
			}
		}
		if = { limit = { tag = ALB } set_country_flag = IR_RfV_ALB_yes } 
		if = { limit = { tag = SER } set_country_flag = IR_RfV_SER_yes } 
		#custom_tooltip = 
	}
	option = {		# We cannot fight the Turks at this time
		name = "IR_flavor_HUN.2.b"
		ai_chance = { 
			factor = 50 
			modifier = {
				factor = 0
				tag = ALB
				war_with = TUR
			}
			modifier = {
				factor = 0
				tag = SER
				is_subject = no
			}
			modifier = {
				factor = 2
				tag = SER
				is_subject_of = TUR
				has_opinion = {
					who = TUR
					value = 0
				}
			}
			modifier = {
				factor = 1.5
				tag = SER
				is_subject_of = TUR
				has_opinion = {
					who = TUR
					value = -50
				}
			}
		}
		if = { limit = { tag = ALB } set_country_flag = IR_RfV_ALB_no } 
		if = { limit = { tag = SER } set_country_flag = IR_RfV_SER_no } 
		#custom_tooltip = 
	}
}
