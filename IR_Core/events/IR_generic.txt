#Calais is lost
country_event = {
	id = IR_generic.1
	title = "IR_generic.1.t"
	desc = "IR_generic.1.desc"
	picture = SIEGE_eventPicture

	fire_only_once = yes

		tag = ENG
		NOT = {
			87 = { owned_by = ROOT }
		}

		months = 1
	}
	
	option = {
		name = "IR_generic.1.a"
	}

		hidden_effect = {
			#remove_country_modifier = IR_england_calais
			87 = { remove_province_modifier = england_calais_modifier }
		}
	}
}
