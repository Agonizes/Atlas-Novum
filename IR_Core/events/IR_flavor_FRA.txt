# Consort Mary Queen of Scots
country_event = {
	id = IR_flavor_FRA.3
	title = "IR_flavor_FRA.3.t"
	desc = "IR_flavor_FRA.3.d"
	picture = COURT_eventPicture

	is_triggered_only = yes
	major = yes
	
	trigger = {
		NOT = { has_country_flag = IR_flavor_FRA.3_happened has_country_flag = IR_flavor_FRA.3_not_happened } 
	}
	
	option = { #accept
		name = "IR_flavor_FRA.3.a" 
		#tooltip = { NAP = { inherit = TAN } }
		ai_chance = { factor = 75 }
		FRA = {	
			define_consort = {
			name = "Mary, Queen of Scots"
			culture = scottish
			adm = 5
			dip = 4
			mil = 5
			female = yes
			}
			set_country_flag = IR_flavor_FRA.3_happened
		}
	}
	option = { #decline
		name = "IR_flavor_FRA.3.b" 
		ai_chance = { factor = 25 }
		set_country_flag = IR_flavor_FRA.3_not_happened
	}
}
