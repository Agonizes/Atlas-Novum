#Elder/Chief/Priestly Council
country_event = {
	id = IR_govt_tribal.1
	title = "IR_govt_tribal.1.t"
	desc = "IR_govt_tribal.1.d"
	picture = COURT_eventPicture

	is_triggered_only = yes

	trigger = {
		government = tribal
		OR = {
			has_reform = chiefs_council
			has_reform = elder_council
			has_reform = priestly_council
		}
	}
	
	immediate = {
		hidden_effect = {
			if = {
				limit = { has_reform = chiefs_council } 
				define_exiled_ruler = {
					exiled_as = tribal_council_candidate_a_@ROOT
					adm = 3
					max_random_adm = 6
					dip = 0
					max_random_dip = 3
					mil = 0
					max_random_mil = 3
					age = 30
					add_ruler_personality = lawgiver_personality
					option = 0
				}
				define_exiled_ruler = {
					exiled_as = tribal_council_candidate_b_@ROOT
					adm = 0
					max_random_adm = 3
					dip = 3
					max_random_dip = 6
					mil = 0
					max_random_mil = 3
					age = 30
					add_ruler_personality = charismatic_negotiator_personality
					option = 1
				}
				define_exiled_ruler = {
					exiled_as = tribal_council_candidate_c_@ROOT
					adm = 0
					max_random_adm = 3
					dip = 0
					max_random_dip = 3
					mil = 3
					max_random_mil = 6
					age = 30
					add_ruler_personality = inspiring_leader_personality
					option = 2
				}
			}
			else_if = {
				limit = { has_reform = elder_council } 
				define_exiled_ruler = {
					exiled_as = tribal_council_candidate_a_@ROOT
					adm = 2
					max_random_adm = 5
					dip = 1
					max_random_dip = 3
					mil = 1
					max_random_mil = 3
					age = 60
					option = 0
				}
				define_exiled_ruler = {
					exiled_as = tribal_council_candidate_b_@ROOT
					adm = 1
					max_random_adm = 3
					dip = 2
					max_random_dip = 5
					mil = 1
					max_random_mil = 3
					age = 60
					option = 1
				}
				define_exiled_ruler = {
					exiled_as = tribal_council_candidate_c_@ROOT
					adm = 1
					max_random_adm = 3
					dip = 1
					max_random_dip = 3
					mil = 2
					max_random_mil = 5
					age = 60
					option = 2
				}
			}
			else_if = { 
				limit = { has_reform = priestly_council }
				define_exiled_ruler = {
					exiled_as = tribal_council_candidate_a_@ROOT
					adm = 1
					max_random_adm = 4
					dip = 0
					max_random_dip = 3
					mil = 0
					max_random_mil = 3
					religion = PREV
					option = 0
				}
				define_exiled_ruler = {
					exiled_as = tribal_council_candidate_b_@ROOT
					adm = 0
					max_random_adm = 3
					dip = 1
					max_random_dip = 4
					mil = 0
					max_random_mil = 3
					religion = PREV
					option = 1
				}
				define_exiled_ruler = {
					exiled_as = tribal_council_candidate_c_@ROOT
					adm = 0
					max_random_adm = 3
					dip = 0
					max_random_dip = 3
					mil = 1
					max_random_mil = 4
					religion = PREV
					option = 2
				}
			}
		}
	}
	
	after = {
		clear_exiled_ruler = tribal_council_candidate_a_@ROOT
		clear_exiled_ruler = tribal_council_candidate_b_@ROOT
		clear_exiled_ruler = tribal_council_candidate_c_@ROOT
	}
	#cheifs council
	option = {		
		name = "IR_govt_tribal.1.a"
		trigger = { has_reform = chiefs_council }
		ai_chance = { factor = 30 }
		set_ruler = tribal_council_candidate_a_@ROOT
	}
	option = {	
		name = "IR_govt_tribal.1.b"
		trigger = { has_reform = chiefs_council } 
		ai_chance = { factor = 30 }
		set_ruler = tribal_council_candidate_b_@ROOT
	}
	option = {	
		name = "IR_govt_tribal.1.c"
		trigger = { has_reform = chiefs_council }
		ai_chance = { factor = 30 }
		set_ruler = tribal_council_candidate_c_@ROOT
	}
	#elder council
	option = {		
		name = "IR_govt_tribal.1.e"
		trigger = { has_reform = elder_council }
		ai_chance = { factor = 30 }
		set_ruler = tribal_council_candidate_a_@ROOT
	}
	option = {	
		name = "IR_govt_tribal.1.f"
		trigger = { has_reform = elder_council }
		ai_chance = { factor = 30 }
		set_ruler = tribal_council_candidate_b_@ROOT
	}
	option = {	
		name = "IR_govt_tribal.1.g"
		trigger = { has_reform = elder_council }
		ai_chance = { factor = 30 }
		set_ruler = tribal_council_candidate_c_@ROOT
	}
	# priestly council
	option = {		
		name = "IR_govt_tribal.1.h"
		trigger = { has_reform = priestly_council }
		ai_chance = { factor = 30 }
		set_ruler = tribal_council_candidate_a_@ROOT
	}
	option = {	
		name = "IR_govt_tribal.1.i"
		trigger = { has_reform = priestly_council }
		ai_chance = { factor = 30 }
		set_ruler = tribal_council_candidate_b_@ROOT
	}
	option = {	
		name = "IR_govt_tribal.1.j"
		trigger = { has_reform = priestly_council }
		ai_chance = { factor = 30 }
		set_ruler = tribal_council_candidate_c_@ROOT
	}
}
